function myFunction() {
 var x = document.getElementById("myText").value;
 var res = x.split(",");
 //Média
 var media = 0;
 for (var i = 0; i < res.length; i++)
 {
  media += parseFloat(res[i]);
 }
 media = media / res.length;
 var desvioPadrao = 0;
 for (var i = 0; i < res.length; i++)
 {
  desvioPadrao += (parseFloat(res[i]) - media) * (parseFloat(res[i]) - media);
 }
 desvioPadrao = Math.sqrt(desvioPadrao / (res.length - 1));

 var desvioPadraoDaMedia = desvioPadrao / Math.sqrt(res.length);
 
 document.getElementById("media").innerHTML = media;
 document.getElementById("desvioPadrao").innerHTML = desvioPadrao;;
 document.getElementById("demo").innerHTML = desvioPadraoDaMedia;
}
